CREATE TABLE EMPLOYEE_PERSONAL(
EMP_ID NUMBER(5) PRIMARY KEY,
EMP_NAME VARCHAR2(20),
ADDRESS VARCHAR2(30),
CITY VARCHAR2(20),
PIN_CODE NUMBER(6)
);
CREATE TABLE EMPLOYEE_SALARY(
EMP_ID NUMBER(5),
SALARY NUMBER(8),
CONSTRAINT FK_EMP
FOREIGN KEY (EMP_ID) REFERENCES
EMPLOYEE_PERSONAL(EMP_ID)
);
